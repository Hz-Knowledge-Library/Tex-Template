% Simple Academic Title Page Layout
% ----------------------------------------------------------

\begin{titlepage}
    \centering
    \vspace*{2cm}
    
    % Title
    {\Huge\bfseries \myTitle \par}
    \vspace{2cm}
    
    % Author
    {\Large \myAuthor \par}
    \vspace{1cm}
    
    % Date
    {\large \myDate \par}
    
    \vfill
\end{titlepage}
